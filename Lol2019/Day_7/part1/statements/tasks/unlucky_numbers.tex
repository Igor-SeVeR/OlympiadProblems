\begin{problem}%
{Несчастливые числа}%
{\textsl{стандартный ввод}}%
{\textsl{стандартный вывод}}%
{1 секунда}%
{64 мегабайта}{}

\InputFile

Вводится число $N$ ($1 \le N \le 71$).

\OutputFile

Выведите $N$-е несчастливое число.

\Examples

\begin{example}
\exmp{
1
}{%
2
}%
\exmp{
2
}{%
4
}%
\end{example}
\end{problem}
